n级阶梯问题

2019-05-22

一个楼梯有N节，每次只能走1步或者2步，请问走完N节楼梯有几种走法？

// 递归结合动态规划
function step1 (n, map) {
  if (n <= 0) return 0
  if (n <= 2) return n
  if (map[n]) return map[n]
  const count = step1(n - 1, map) + step1(n-2, map)
  map[n] = count
  return count
}

// 动态规划
function step2 (n) {
  if (n <= 0) return 0
  if (n <= 2) return n
  const a = new Array(10000).fill(0)
  a[1] = 1
  a[2] = 2
  let i
  for (i = 3;i <= n;i++) {
    a[i] = a[i - 1] + a[i -2]
  }
  return a[i - 1]
}

// 节省空间的动态规划
function step3 (n) {
  if (n <= 0) return 0
  if (n <= 2) return n
  let i= 3, a1 = 1,a2 = 2
  while (i <= n ) {
    let temp = a1 + a2
    a1 = a2
    a2 = temp
    i++
  }
  return a2
}

console.log(step1(4, {}), step2(4), step3(4))
